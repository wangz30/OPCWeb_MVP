## 📍 关键跳转代码位置

### 导航栏配置

**文件**: `src/components/Navigation.tsx`

```typescript
const navItems = [
  { path: '/', label: '首页', icon: Home },
  { 
    path: '/products', 
    label: '产品', 
    icon: Sparkles,
    children: [
      { path: '/models', label: 'AI 模型', icon: Cpu },
      { path: '/data-apis', label: '数据 API', icon: Database },
      { path: '/cloud-resources', label: '云资源', icon: Cloud },
    ]
  },
  // ... 其他菜单项
]
```

---

### 路由配置

**文件**: `src/App.tsx`

```typescript
<Routes>
  {/* 后台管理 - 独立布局 */}
  <Route path="/dashboard" element={<DashboardPage />} />
  
  {/* 前台页面 - 带导航 */}
  <Route path="/*" element={
    <div>
      <Navigation />
      <main>
        <Routes>
          <Route path="/" element={<HomePage />} />
          <Route path="/products" element={<ProductsPage />} />
          <Route path="/models" element={<ModelsPage />} />
          {/* ... 其他路由 */}
        </Routes>
      </main>
    </div>
  } />
</Routes>
```
